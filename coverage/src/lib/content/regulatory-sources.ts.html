export const guidanceLinks = Object.fromEntries(
  Object.entries(REG_SOURCES)

export const findGuidanceLink = (sourceCode: string | null | undefined) => {
  if (!sourceCode) return null;
  const code = sourceCode.trim();
  if (guidanceLinks[code]) return guidanceLinks[code];
  const baseCode = code.split(' ')[0].split('§')[0].trim();
  if (guidanceLinks[baseCode]) return guidanceLinks[baseCode];
  if (code.includes('Deciding When to Submit a 510(k)') || code.includes('Applied by analogy from Deciding'))
    return guidanceLinks['FDA-SW-510K-2017'];
  if (
    code.includes('Marketing Submission Recommendations') ||
    code.includes('PCCP for AI-Enabled') ||
    code.includes('PCCP for Artificial Intelligence')
  )
    return guidanceLinks['FDA-PCCP-2025'];
  if (
    code.includes('AI-Enabled Device Software Functions') ||
    code.includes('AI-DSF') ||
    code.includes('Lifecycle Management')
  )
    return guidanceLinks['FDA-LIFECYCLE-2025'];
  if (code.includes('Cybersecurity in Medical Devices') || code.includes('cybersecurity'))
    return guidanceLinks['FDA-CYBER-2026'];
  if (code.includes('PMA Supplements Guidance')) return guidanceLinks['21 CFR 814.39'];
  if (
    code.includes('Internal reassessment') ||
    code.includes('cumulative') ||
    code.includes('substantial equivalence reassessment')
  )
    return guidanceLinks['FDA-SW-510K-2017'];
  if (code.includes('GenAI risk assessment') || code.includes('GenAI') || code.includes('risk control evaluation'))
    return guidanceLinks['FDA-LIFECYCLE-2025'];
  if (
    code.includes('De Novo') &&
    (code.includes('device-type') || code.includes('special-controls') || code.includes('classification order'))
  )
    return guidanceLinks['21 CFR Part 860'];
  if (code.includes('21 CFR 807')) return guidanceLinks['21 CFR 807'];
  if (code.includes('21 CFR 814.39')) return guidanceLinks['21 CFR 814.39'];
  if (code.includes('21 CFR 814.84')) return guidanceLinks['21 CFR 814.84'];
  if (code.includes('21 CFR') && code.includes('820')) return guidanceLinks['21 CFR Part 820'];
  if (code.includes('ISO 14971')) return guidanceLinks['ISO 14971:2019'];
  if (code.includes('ISO 13485')) return guidanceLinks['ISO 13485:2016'];
  if (code.includes('IEC 62304')) return guidanceLinks['IEC 62304'];
  if (code.includes('MDCG 2022')) return guidanceLinks['MDCG-2022-6'];
  if (code.includes('MDCG 2020')) return guidanceLinks['MDCG-2020-3'];
  if (code.includes('MDCG 2025')) return guidanceLinks['MDCG 2025-6'];
  if (code.includes('MDR')) return guidanceLinks['EU MDR 2017/745'];
  if (code.includes('AI Act')) return guidanceLinks['EU AI Act'];
  if (code.includes('524B')) return guidanceLinks['FD&C 524B'];
  if (code.includes('FDORA') || code.includes('515C')) return guidanceLinks['FDORA 515C'];
  if (code.includes('Q-Sub')) return guidanceLinks['FDA Q-Sub'];
  if (code === 'QMSR' || code.startsWith('QMSR ')) return guidanceLinks['21 CFR Part 820'];
  return null;
};
